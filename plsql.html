PL/SQL Block Structure

Example: 


      DECLARE
      v_count INT;
  BEGIN
      SELECT COUNT(*) INTO v_count FROM employees;
      DBMS_OUTPUT.PUT_LINE('Total number of employees: ' || v_count);
  END;
Output: 

      Total number of employees: X

PL/SQL Data Types, Variables, and Constants

Example: 


      DECLARE
      v_employee_name employees.name%TYPE := 'John';
      v_employee_age INT := 30;
      c_bonus DECIMAL(10, 2) := 500.00;
  BEGIN
      DBMS_OUTPUT.PUT_LINE('Employee Name: ' || v_employee_name);
      DBMS_OUTPUT.PUT_LINE('Employee Age: ' || v_employee_age);
      DBMS_OUTPUT.PUT_LINE('Bonus Amount: ' || c_bonus);
  END;
Output: 

      Employee Name: John
Employee Age: 30
Bonus Amount: 500.00

Scope and Visibility of a Variable

Example: 


      DECLARE
      v_employee_name VARCHAR(50);
  BEGIN
      SELECT name INTO v_employee_name FROM employees WHERE id = 1;
      DBMS_OUTPUT.PUT_LINE('Employee Name: ' || v_employee_name);
  END;
  /
Output: 

      Employee Name: [Name of employee with ID 1]

Assignments and Expressions

Example: 


      DECLARE
      v_total_salary DECIMAL(10, 2);
  BEGIN
      SELECT SUM(salary) INTO v_total_salary FROM employees;
      DBMS_OUTPUT.PUT_LINE('Total Salary: ' || v_total_salary);
  END;
Output: 

      Total Salary: [Sum of all salaries in the employees table]

Conditional and Iterative Control

Example: 


      DECLARE
      v_bonus DECIMAL(10, 2);
  BEGIN
      FOR emp_rec IN (SELECT * FROM employees) LOOP
          IF emp_rec.salary > 50000 THEN
              v_bonus := emp_rec.salary * 0.1;
              DBMS_OUTPUT.PUT_LINE('Bonus for ' || emp_rec.name || ': ' || v_bonus);
          END IF;
      END LOOP;
  END;
Output: 

      Bonus for [Employee with salary > 50000]: [Calculated bonus amount]

Cursor Management in PL/SQL

Example: 


      DECLARE
      CURSOR emp_cursor IS SELECT * FROM employees;
      v_employee employees%ROWTYPE;
  BEGIN
      OPEN emp_cursor;
      LOOP
          FETCH emp_cursor INTO v_employee;
          EXIT WHEN emp_cursor%NOTFOUND;
          DBMS_OUTPUT.PUT_LINE('Employee Name: ' || v_employee.name || ', Salary: ' || v_employee.salary);
      END LOOP;
      CLOSE emp_cursor;
  END;
Output: 

      Employee Name: [Name of employee], Salary: [Salary of employee]

Exception Handling in PL/SQL

Example: 


      DECLARE
      v_salary employees.salary%TYPE;
  BEGIN
      SELECT salary INTO v_salary FROM employees WHERE id = 1000;
      DBMS_OUTPUT.PUT_LINE('Salary of Employee with ID 1000: ' || v_salary);
  EXCEPTION
      WHEN NO_DATA_FOUND THEN
          DBMS_OUTPUT.PUT_LINE('Employee with ID 1000 not found.');
  END;
  /
Output: 

      Employee with ID 1000 not found.
